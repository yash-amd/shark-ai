21@cat.override(AllOfType(Tensor, PrimitiveTensor)) 

22def cat_cpu_eager_f8( 

23 tensors: Sequence[Tensor | PrimitiveTensor], dim: int 

24) -> Tensor | PrimitiveTensor: 

25 """A workaround for torch not having CPU implementation for f8. 

26 During export we don't want to bitcast to int8 as the IREE compiler can't optimize 

27 out the bitcast in some cases.""" 

28 is_inference_tensor = isinstance(tensors[0], PrimitiveTensor) 

29 tensors = [unbox_tensor(t) for t in tensors] 

30 

31 int8_like_dtypes = [torch.float8_e4m3fn, torch.float8_e4m3fnuz] 

32 if torch.compiler.is_compiling() or not all( 

33 t.dtype in int8_like_dtypes and t.is_cpu for t in tensors 

34 ): 

35 return NotImplemented 

36 

37 orig_dtype = tensors[0].dtype 

38 tensors = [t.view(dtype=torch.int8) for t in tensors] 

39 result = cat(tensors, dim) 

40 result = result.view(dtype=orig_dtype) 

41 

42 if is_inference_tensor: 

43 result = DefaultPrimitiveTensor(data=result) 

44 return result
